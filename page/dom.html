function ElementosDom() {
  // Imprimir mensajes iniciales
  console.log("dev");
  console.log("********* Elementos del Dom ***********");

  // Imprimir varios elementos y propiedades del DOM
  console.log("Window Object:", window);
  console.log("Document Object:", document);
  console.log("Head Element:", document.head);
  console.log("Body Element:", document.body);
  console.log("HTML Element:", document.documentElement);
  console.log("Doctype:", document.doctype);
  console.log("Character Set:", document.characterSet);
  console.log("Document Title:", document.title);
  console.log("Links:", document.links);
  console.log("Images:", document.images);
  console.log("Forms:", document.forms);
  console.log("StyleSheets:", document.styleSheets);
  console.log("Scripts:", document.scripts);

  // Esperar 5 segundos y luego imprimir el texto seleccionado
  setTimeout(() => {
    console.log("Selected Text:", document.getSelection().toString());
  }, 5000);
}

export default ElementosDom;
